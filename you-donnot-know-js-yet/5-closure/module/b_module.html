          function CoolModule(){
            var something = "cool";
            var another = [1,2,3];

            function doSomething(){
              console.log(something);
            }

            function doAnother(){
              console.log(another.join("!"));
            }

            return {
              doSomething:doSomething,
              doAnother:doAnother
            }
          })();

          foo.doSomething();
          foo.doAnother();

闭包模式之模块1中每次调用CoolModule()都会创建一个模块实例，当只需要一个模块实例时，可以通过将模块函数转换成IIFE来实现
